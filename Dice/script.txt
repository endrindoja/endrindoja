void setup() {
  Serial.begin(9600); 
  pinMode(7, INPUT_PULLUP);

  display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDR);
  display.clearDisplay();
  delay(10); 

  randomSeed(analogRead(0));
}

void loop() {
  
if(LOW == digitalRead(7)){
 randNumber = random(1,7);

if(randNumber == 1){
  display.clearDisplay();
  display.drawBitmap(35, 0, epd_bitmap_1, 128, 64, WHITE);
  display.display();

}

if(randNumber == 2){
  display.clearDisplay();
  display.drawBitmap(35, 0, epd_bitmap_2, 128, 64, WHITE);
  display.display();
}

  display.display();
}

if(randNumber == 5){
  display.clearDisplay();

}
